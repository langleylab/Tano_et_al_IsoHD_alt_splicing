    intronjunctionsintersectlistlong %>%

        y <- unlist(x[,2:3])
        rename <- setNames(object = seq_len(length(unique(sort(y)))),
                           nm = unique(sort(y)))
        x[,2:3] <- matrix(data = rename[as.character(y)], nrow = nrow(x))
        x
    })
